from typing import Dict, List

class AccessControl:
    def __init__(self):
        # N√≠veis de permiss√£o e autentica√ß√£o
        self.roles = {
            "admin": ["multifactor_auth", "full_access"],
            "developer": ["multifactor_auth", "limited_access"],
            "user": ["basic_auth", "restricted_access"]
        }
        self.active_sessions = {}  # Sess√µes ativas de usu√°rios

    def authenticate(self, user_id: str, role: str, auth_method: str) -> bool:
        """
        Autentica um usu√°rio com base no m√©todo de autentica√ß√£o e fun√ß√£o.

        Args:
            user_id (str): ID do usu√°rio.
            role (str): Fun√ß√£o do usu√°rio (admin, developer, user).
            auth_method (str): M√©todo de autentica√ß√£o utilizado.

        Returns:
            bool: True se a autentica√ß√£o for bem-sucedida, False caso contr√°rio.
        """
        if role not in self.roles:
            return False
        if auth_method not in self.roles[role]:
            return False
        self.active_sessions[user_id] = role
        return True

    def check_permission(self, user_id: str, permission: str) -> bool:
        """
        Verifica se o usu√°rio tem permiss√£o para realizar uma a√ß√£o.

        Args:
            user_id (str): ID do usu√°rio.
            permission (str): Permiss√£o requerida.

        Returns:
            bool: True se o usu√°rio tiver permiss√£o, False caso contr√°rio.
        """
        role = self.active_sessions.get(user_id)
        if not role:
            return False
        return permission in self.roles[role]

    def logout(self, user_id: str) -> None:
        """
        Encerra a sess√£o de um usu√°rio.

        Args:
            user_id (str): ID do usu√°rio.
        """
        if user_id in self.active_sessions:
            del self.active_sessions[user_id]

def extrair_diretrizes(texto: str) -> List[Dict]:
    """
    Extrai diretrizes de um texto usando NLP.
    """
    nlp = pipeline("question-answering")
    perguntas = [
        "Qual √© o ID da diretriz?",
        "Qual √© o enunciado da diretriz?",
        "Qual √© o tipo da diretriz?",
        "Quem √© aplic√°vel?",
        "Qual √© a justificativa?"
    ]
    diretrizes = []
    for pergunta in perguntas:
        resposta = nlp(question=pergunta, context=texto)
        diretrizes.append(resposta["answer"])
    return diretrizes

def validar_modulo(modulo: Dict) -> bool:
    """
    Valida a estrutura de um m√≥dulo de conhecimento.
    """
    campos_obrigatorios = ["modulo", "categoria", "subcategoria", "descricao", "diretrizes", "palavras_chave", "fonte"]
    for campo in campos_obrigatorios:
        if campo not in modulo:
            print(f"Erro: Campo obrigat√≥rio '{campo}' ausente.")
            return False
    return True
